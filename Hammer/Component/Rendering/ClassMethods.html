- (Object) define_widget(name = :Widget, parent = nil) { ... }

Deprecated.

widgets can be now normally defined with class

block is used to define #content not class in parent components. Class is used directly. see #parent_widget_class

Examples:

widget with name Header 

define_widget :Header do
  wrap_in :h1
  def content
    text "A header"
  end
end

quickly defined widget with default name Widget 

define_widget :quickly do
  def content
    h1 "A header"
  end
end

Parameters:

  • (Symbol) name (defaults to: :Widget) —

    of a new widget class. If :quicly is passed defines widget quickly which means that

  • (Symbol, Class) parent (defaults to: nil) —

    class of new widget class. Symbol is used to find widget class with same name

Yields:

  • block which is evaluated inside new widget class or #contet if :quicly is used

def define_widget(name = :Widget, parent = nil, &block) # TODO remove method
  Hammer.logger.warn 'define_widget is deprecated'
  if name == :quickly
    define_widget { define_method :content, &block }
  else
    parent = parent_widget_class(parent || name)
    widget_class = const_set(name, Class.new(parent))
    # extend_widget(widget_class)
    widget_class.class_eval(&block)
  end
end

def extend_widgets
  widget_classes(false).values.each do |widget_class|
    extend_widget_by.each do |a_module|
      unless widget_class.include? a_module
        widget_class.send :include, a_module
      end
    end
  end
end

def parent_widget_class(name)
  return name if name.kind_of? Class
  check_class widget_classes[name] || begin
    superclass.parent_widget_class name if superclass.respond_to? :parent_widget_class
  end
end

def widget_class_candidates(inherited)
  if Hammer.v19?
    self.constants(inherited)
  else
    self.constants + if inherited && superclass.respond_to?(:widget_class_candidates, true)
      superclass.send :widget_class_candidates, inherited
    else
      []
    end
  end.map(&:to_sym)
end

def widget_classes(inherited = true)
  widget_class_candidates(inherited).inject({}) do |hash, const|
    if (klass = const_get(const)) < Hammer::Widget::Base
      hash[const] = klass
    end
    hash
  end
end
