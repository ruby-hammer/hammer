def self.included(base)
  base.extend ClassMethods
  base.class_inheritable_accessor :_on_submit, :instance_reader => false, :instance_writer => false
  base.class_eval do
    needs :record
    attr_reader :record
    changing :set_value
  end
end

def set_value(*args)
  key, value = args.pop 2
  unless value
    value, key = key, :value
  end
  @record.send "#{key}=", value
end
