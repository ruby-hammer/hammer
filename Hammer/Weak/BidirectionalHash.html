def add(key, value)
  @hash[key] = value
  @reverse[value] ||= []
  @reverse[value] << key
end

def delete(key)
  if value = @hash[key]
    @reverse[value].delete(key)
    @reverse.delete value if @reverse[value].empty?
    @hash.delete key
  end
end

def delete_value(value)
  keys = @reverse[value]
  @reverse.delete value
  keys.each {|key| @hash.delete(key) }
end
