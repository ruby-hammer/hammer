def self.finalize_itself

def get_key_value(key)
  if v = storage.get(key.object_id)
    return key.object_id, v
  end
  storage.get_candidates(key.hash).find do |k,v|
    candidate_key = get_object(k) or next
    candidate_key.eql? key
  end
end
