def execute_non_query(*args)
  f = Fiber.current
  return super if f == MAIN_FIBER

  EventMachine.defer(
    proc do
      begin
        super
      rescue Exception => e
        e
      end
    end,
    proc do |r|
      f.resume(r)
    end
  )

  r = Fiber.yield
  if r.kind_of?(Exception)
    r.set_backtrace(caller)
    raise r
  end
  r
end
