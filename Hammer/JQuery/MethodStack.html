def _method(method, *args)
  method.to_s =~ /([\w_$]+)(!|)/
  method = ALIASES[$1] || $1
  if args.blank? && $2.blank?
    @methods << method
  else
    @methods << "#{method}(#{_format_args(*args)})"
  end
  self
end
