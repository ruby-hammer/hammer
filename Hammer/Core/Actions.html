def register_action(component, &block)
  uuid = Hammer::Core.generate_id
  @actions[uuid] = Action.new(uuid, component, block)
  return uuid
end

def run_action(id, arg)
  return self unless action = @actions[id]
  action.call(component_by_id(arg))
  self
end
