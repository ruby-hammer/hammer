def self.benchmark(label, req = true, &block)
  time = Benchmark.realtime { block.call }
  Hammer.logger.info "#{label} in %0.6f sec" % time unless req
  Hammer.logger.info "#{label} in %0.6f sec ~ %d req" % [time, (1/time).to_i] if req
end
